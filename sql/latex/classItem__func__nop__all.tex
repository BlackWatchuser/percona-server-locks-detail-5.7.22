     a < b              -> a >= b
     a >= b             -> a < b
     a > b              -> a <= b
     a <= b             -> a > b
     IS NULL(a)         -> IS NOT NULL(a)
     IS NOT NULL(a)     -> IS NULL(a)\end{DoxyVerb}
