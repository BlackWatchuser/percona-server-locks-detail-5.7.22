156  /*
157  Internal states for parsing a stream of events.
158 
159  DDL has the format:
160  DDL-1: [GTID]
161  DDL-2: [User] [Intvar] [Rand]
162  DDL-3: Query
163 
164  DML has the format:
165  DML-1: [GTID]
166  DML-2: Query(BEGIN)
167  DML-3: Statements
168  DML-4: (Query(COMMIT) | Query([XA] ROLLBACK) | Xid | Xa_prepare)
169  */
170  enum enum_event_parser_state {
171  /* NONE is set after DDL-3 or DML-4 */
172  EVENT_PARSER_NONE,
173  /* GTID is set after DDL-1 or DML-1 */
174  EVENT_PARSER_GTID,
175  /* DDL is set after DDL-2 */
176  EVENT_PARSER_DDL,
177  /* DML is set after DML-2 */
178  EVENT_PARSER_DML,
179  /* ERROR is set whenever the above pattern is not followed */
180  EVENT_PARSER_ERROR
181  };
