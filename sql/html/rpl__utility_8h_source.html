254  /*
255  Returns internal binlog type code for one field,
256  without translation to real types.
257  */
258  enum_field_types binlog_type(ulong index) const
259  {
260  return static_cast<enum_field_types>(m_type[index]);
261  }
262  /*
263  Return a representation of the type data for one field.
264 
265  @param index Field index to return data for
266 
267  @return Will return a representation of the type data for field
268  <code>index</code>. Currently, only the type identifier is
269  returned.
270  */
271  enum_field_types type(ulong index) const
272  {
273  DBUG_ASSERT(index < m_size);
274  /*
275  If the source type is MYSQL_TYPE_STRING, it can in reality be
276  either MYSQL_TYPE_STRING, MYSQL_TYPE_ENUM, or MYSQL_TYPE_SET, so
277  we might need to modify the type to get the real type.
278  */
279  enum_field_types source_type= binlog_type(index);
280  uint16 source_metadata= m_field_metadata[index];
281  switch (source_type)
282  {
283  case MYSQL_TYPE_STRING:
284  {
285  int real_type= source_metadata >> 8;
286  if (real_type == MYSQL_TYPE_ENUM || real_type == MYSQL_TYPE_SET)
287  source_type= static_cast<enum_field_types>(real_type);
288  break;
289  }
290 
291  /*
292  This type has not been used since before row-based replication,
293  so we can safely assume that it really is MYSQL_TYPE_NEWDATE.
294  */
295  case MYSQL_TYPE_DATE:
296  source_type= MYSQL_TYPE_NEWDATE;
297  break;
298 
299  default:
300  /* Do nothing */
301  break;
302  }
303 
304  return source_type;
305  }

308  /*
309  This function allows callers to get the extra field data from the
310  table map for a given field. If there is no metadata for that field
311  or there is no extra metadata at all, the function returns 0.
312 
313  The function returns the value for the field metadata for column at
314  position indicated by index. As mentioned, if the field was a type
315  that stores field metadata, that value is returned else zero (0) is
316  returned. This method is used in the unpack() methods of the
317  corresponding fields to properly extract the data from the binary log
318  in the event that the master's field is smaller than the slave.
319  */
320  uint16 field_metadata(uint index) const

322  DBUG_ASSERT(index < m_size);
