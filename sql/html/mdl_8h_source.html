152 enum enum_mdl_type {
153  /*
154  An intention exclusive metadata lock. Used only for scoped locks.
155  Owner of this type of lock can acquire upgradable exclusive locks on
156  individual objects.
157  Compatible with other IX locks, but is incompatible with scoped S and
158  X locks.
159  */
160  MDL_INTENTION_EXCLUSIVE= 0,
161  /*
162  A shared metadata lock.
163  To be used in cases when we are interested in object metadata only
164  and there is no intention to access object data (e.g. for stored
165  routines or during preparing prepared statements).
166  We also mis-use this type of lock for open HANDLERs, since lock
167  acquired by this statement has to be compatible with lock acquired
168  by LOCK TABLES ... WRITE statement, i.e. SNRW (We can't get by by
169  acquiring S lock at HANDLER ... OPEN time and upgrading it to SR
170  lock for HANDLER ... READ as it doesn't solve problem with need
171  to abort DML statements which wait on table level lock while having
172  open HANDLER in the same connection).
173  To avoid deadlock which may occur when SNRW lock is being upgraded to
174  X lock for table on which there is an active S lock which is owned by
175  thread which waits in its turn for table-level lock owned by thread
176  performing upgrade we have to use thr_abort_locks_for_thread()
177  facility in such situation.
178  This problem does not arise for locks on stored routines as we don't
179  use SNRW locks for them. It also does not arise when S locks are used
180  during PREPARE calls as table-level locks are not acquired in this
181  case.
182  */
183  MDL_SHARED,
184  /*
185  A high priority shared metadata lock.
186  Used for cases when there is no intention to access object data (i.e.
187  data in the table).
188  "High priority" means that, unlike other shared locks, it is granted
189  ignoring pending requests for exclusive locks. Intended for use in
190  cases when we only need to access metadata and not data, e.g. when
191  filling an INFORMATION_SCHEMA table.
192  Since SH lock is compatible with SNRW lock, the connection that
193  holds SH lock lock should not try to acquire any kind of table-level
194  or row-level lock, as this can lead to a deadlock. Moreover, after
195  acquiring SH lock, the connection should not wait for any other
196  resource, as it might cause starvation for X locks and a potential
197  deadlock during upgrade of SNW or SNRW to X lock (e.g. if the
198  upgrading connection holds the resource that is being waited for).
199  */
200  MDL_SHARED_HIGH_PRIO,
201  /*
202  A shared metadata lock for cases when there is an intention to read data
203  from table.
204  A connection holding this kind of lock can read table metadata and read
205  table data (after acquiring appropriate table and row-level locks).
206  This means that one can only acquire TL_READ, TL_READ_NO_INSERT, and
207  similar table-level locks on table if one holds SR MDL lock on it.
208  To be used for tables in SELECTs, subqueries, and LOCK TABLE ... READ
209  statements.
210  */
211  MDL_SHARED_READ,
212  /*
213  A shared metadata lock for cases when there is an intention to modify
214  (and not just read) data in the table.
215  A connection holding SW lock can read table metadata and modify or read
216  table data (after acquiring appropriate table and row-level locks).
217  To be used for tables to be modified by INSERT, UPDATE, DELETE
218  statements, but not LOCK TABLE ... WRITE or DDL). Also taken by
219  SELECT ... FOR UPDATE.
220  */
221  MDL_SHARED_WRITE,
222  /*
223  A version of MDL_SHARED_WRITE lock which has lower priority than
224  MDL_SHARED_READ_ONLY locks. Used by DML statements modifying
225  tables and using the LOW_PRIORITY clause.
226  */
227  MDL_SHARED_WRITE_LOW_PRIO,
228  /*
229  An upgradable shared metadata lock which allows concurrent updates and
230  reads of table data.
231  A connection holding this kind of lock can read table metadata and read
232  table data. It should not modify data as this lock is compatible with
233  SRO locks.
234  Can be upgraded to SNW, SNRW and X locks. Once SU lock is upgraded to X
235  or SNRW lock data modification can happen freely.
236  To be used for the first phase of ALTER TABLE.
237  */
238  MDL_SHARED_UPGRADABLE,
239  /*
240  A shared metadata lock for cases when we need to read data from table
241  and block all concurrent modifications to it (for both data and metadata).
242  Used by LOCK TABLES READ statement.
243  */
244  MDL_SHARED_READ_ONLY,
245  /*
246  An upgradable shared metadata lock which blocks all attempts to update
247  table data, allowing reads.
248  A connection holding this kind of lock can read table metadata and read
249  table data.
250  Can be upgraded to X metadata lock.
251  Note, that since this type of lock is not compatible with SNRW or SW
252  lock types, acquiring appropriate engine-level locks for reading
253  (TL_READ* for MyISAM, shared row locks in InnoDB) should be
254  contention-free.
255  To be used for the first phase of ALTER TABLE, when copying data between
256  tables, to allow concurrent SELECTs from the table, but not UPDATEs.
257  */
258  MDL_SHARED_NO_WRITE,
259  /*
260  An upgradable shared metadata lock which allows other connections
261  to access table metadata, but not data.
262  It blocks all attempts to read or update table data, while allowing
263  INFORMATION_SCHEMA and SHOW queries.
264  A connection holding this kind of lock can read table metadata modify and
265  read table data.
266  Can be upgraded to X metadata lock.
267  To be used for LOCK TABLES WRITE statement.
268  Not compatible with any other lock type except S and SH.
269  */
270  MDL_SHARED_NO_READ_WRITE,
271  /*
272  An exclusive metadata lock.
273  A connection holding this lock can modify both table's metadata and data.
274  No other type of metadata lock can be granted while this lock is held.
275  To be used for CREATE/DROP/RENAME TABLE statements and for execution of
276  certain phases of other DDL statements.
277  */
278  MDL_EXCLUSIVE,
279  /* This should be the last !!! */
280  MDL_TYPE_END};

949  void set_needs_thr_lock_abort(bool needs_thr_lock_abort)
950  {
951  /*
952  @note In theory, this member should be modified under protection
953  of some lock since it can be accessed from different threads.
954  In practice, this is not necessary as code which reads this
955  value and so might miss the fact that value was changed will
956  always re-try reading it after small timeout and therefore
957  will see the new value eventually.
958  */
