30  there is no limit.

32 typedef struct user_resources {

34  uint questions;

36  Maximum number of updating statements per hour (which statements are

38  */

41  uint conn_per_hour;
42  /* Maximum number of concurrent connections. */
43  uint user_conn;
44  /*
45  Values of this enum and specified_limits member are used by the

47  */
48  enum {QUERIES_PER_HOUR= 1, UPDATES_PER_HOUR= 2, CONNECTIONS_PER_HOUR= 4,
49  USER_CONNECTIONS= 8};
50  uint specified_limits;
51 } USER_RESOURCES;

60  Pointer to user+host key (pair separated by '\0') defining the entity

75  /* Current amount of concurrent connections for this account. */
76  uint connections;

78  Current number of connections per hour, number of updating statements

83  USER_RESOURCES user_resources;
