30 /*
31  System variables controlling logging:
32 
33  log_output (--log-output)
34  Values: NONE, FILE, TABLE
35  Select output destination. Does not enable logging.
36  Can set more than one (e.g. TABLE | FILE).
37 
38  general_log (--general_log)
39  slow_query_log (--slow_query_log)
40  Values: 0, 1
41  Enable/disable general/slow query log.
42 
43  general_log_file (--general-log-file)
44  slow_query_log_file (--slow-query-log-file)
45  Values: filename
46  Set name of general/slow query log file.
47 
48  sql_log_off
49  Values: ON, OFF
50  Enable/disable general query log (OPTION_LOG_OFF).
51 
52  log_queries_not_using_indexes (--log-queries-not-using-indexes)
53  Values: ON, OFF
54  Control slow query logging of queries that do not use indexes.
55 
56  --log-raw
57  Values: ON, OFF
58  Control query rewrite of passwords to the general log.
59 
60  --log-short-format
61  Values: ON, OFF
62  Write short format to the slow query log (and the binary log).
63 
64  --log-slow-admin-statements
65  Values: ON, OFF
66  Log statements such as OPTIMIZE TABLE, ALTER TABLE to the slow query log.
67 
68  --log-slow-slave-statements
69  Values: ON, OFF
70 
71  log_throttle_queries_not_using_indexes
72  Values: INT
73  Number of queries not using indexes logged to the slow query log per min.
74 */
