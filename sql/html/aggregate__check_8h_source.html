497  bool is_stopped(const Item *i)
498  {
499  if (stopped_at_item)
500  {
501  /*
502  Walking was disabled for a tree part rooted a one ancestor of 'i' or
503  rooted at 'i'.
504  */
505  if (stopped_at_item == i)
506  {
507  /*
508  Walking was disabled for the tree part rooted at 'i'; we have now just
509  returned back to this root (WALK_POSTFIX call), left the tree part:
510  enable the walk again, for other tree parts.
511  */
512  stopped_at_item= NULL;
513  }
514  // No further processing to do for this item:
515  return true;
516  }
517  return false;
518  }
