117 class Query_result_explain : public Query_result_send {
118 protected:
119  /*
120  As far as we use Query_result_explain object in a place of Query_result_send,
121  Query_result_explain have to pass multiple invocation of its prepare(),
122  prepare2() and initialize_tables() functions, since JOIN::exec() of
123  subqueries runs these functions of Query_result_send multiple times by design.
124  Query_result_insert, Query_result_update and Query_result_delete class
125  functions are not intended for multiple invocations, so "prepared",
126  "prepared2" and "initialized" flags guard data interceptor object from
127  function re-invocation.
128  */
129  bool prepared;
130  bool prepared2;
131  bool initialized;
132 
137  Query_result *interceptor;
138 
139 public:
140  Query_result_explain(st_select_lex_unit *unit_arg, Query_result *interceptor_arg)
141  : prepared(false), prepared2(false), initialized(false),
142  interceptor(interceptor_arg)
143  { unit= unit_arg; }
144 
145 protected:
146  virtual int prepare(List<Item> &list, SELECT_LEX_UNIT *u)
147  {
148  if (prepared)
149  return false;
150  prepared= true;
151  return Query_result_send::prepare(list, u) || interceptor->prepare(list, u);
152  }
153 
154  virtual int prepare2(void)
155  {
156  if (prepared2)
157  return false;
158  prepared2= true;
159  return Query_result_send::prepare2() || interceptor->prepare2();
160  }
161 
162  virtual bool initialize_tables(JOIN *join)
163  {
164  if (initialized)
165  return false;
166  initialized= true;
167  return Query_result_send::initialize_tables(join) ||
168  interceptor->initialize_tables(join);
169  }
170 
171  virtual void cleanup()
172  {
173  Query_result_send::cleanup();
174  interceptor->cleanup();
175  }
176 };
