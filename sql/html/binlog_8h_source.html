477  /*
478  Holds the last seen in Relay-Log FD's checksum alg value.
479  The initial value comes from the slave's local FD that heads
480  the very first Relay-Log file. In the following the value may change
481  with each received master's FD_m.
482  Besides to be used in verification events that IO thread receives
483  (except the 1st fake Rotate, see @c Master_info:: checksum_alg_before_fd),
484  the value specifies if/how to compute checksum for slave's local events
485  and the first fake Rotate (R_f^1) coming from the master.
486  R_f^1 needs logging checksum-compatibly with the RL's heading FD_s.
487 
488  Legends for the checksum related comments:
489 
490  FD - Format-Description event,
491  R - Rotate event
492  R_f - the fake Rotate event
493  E - an arbirary event
494 
495  The underscore indexes for any event
496  `_s' indicates the event is generated by Slave
497  `_m' - by Master
498 
499  Two special underscore indexes of FD:
500  FD_q - Format Description event for queuing (relay-logging)
501  FD_e - Format Description event for executing (relay-logging)
502 
503  Upper indexes:
504  E^n - n:th event is a sequence
505 
506  RL - Relay Log
507  (A) - checksum algorithm descriptor value
508  FD.(A) - the value of (A) in FD
509  */

1029 inline bool normalize_binlog_name(char *to, const char *from, bool is_relay_log)
1030 {
1031  DBUG_ENTER("normalize_binlog_name");
1032  bool error= false;
1033  char buff[FN_REFLEN];
1034  char *ptr= (char*) from;
1035  char *opt_name= is_relay_log ? opt_relay_logname : opt_bin_logname;
1036 
1037  DBUG_ASSERT(from);
1038 
1039  /* opt_name is not null and not empty and from is a relative path */
1040  if (opt_name && opt_name[0] && from && !test_if_hard_path(from))
1041  {
1042  // take the path from opt_name
1043  // take the filename from from
1044  char log_dirpart[FN_REFLEN], log_dirname[FN_REFLEN];
1045  size_t log_dirpart_len, log_dirname_len;
1046  dirname_part(log_dirpart, opt_name, &log_dirpart_len);
1047  dirname_part(log_dirname, from, &log_dirname_len);
1048 
1049  /* log may be empty => relay-log or log-bin did not
1050  hold paths, just filename pattern */
1051  if (log_dirpart_len > 0)
1052  {
1053  /* create the new path name */
1054  if(fn_format(buff, from+log_dirname_len, log_dirpart, "",
1055  MYF(MY_UNPACK_FILENAME | MY_SAFE_PATH)) == NULL)
1056  {
1057  error= true;
1058  goto end;
1059  }
1060 
1061  ptr= buff;
1062  }
1063  }
1064 
1065  DBUG_ASSERT(ptr);
1066  if (ptr)
1067  {
1068  size_t length= strlen(ptr);
1069 
1070  // Strips the CR+LF at the end of log name and \0-terminates it.
1071  if (length && ptr[length-1] == '\n')
1072  {
1073  ptr[length-1]= 0;
1074  length--;
1075  if (length && ptr[length-1] == '\r')
1076  {
1077  ptr[length-1]= 0;
1078  length--;
1079  }
1080  }
1081  if (!length)
1082  {
1083  error= true;
1084  goto end;
1085  }
1086  strmake(to, ptr, length);
1087  }
