131 /*
132  Call a BG function with specified types of operands. We have to create
133  geo1 and geo2 because operands g1 and g2 are created without their WKB data
134  parsed, so not suitable for BG to use. geo1 will share the same copy of WKB
135  data with g1, also true for geo2.
136  */
137 #define BGCALL(res, bgfunc, GeoType1, g1, GeoType2, g2, pnullval) do { \
138  const void *pg1= g1->normalize_ring_order(); \
139  const void *pg2= g2->normalize_ring_order(); \
140  if (pg1 != NULL && pg2 != NULL) \
141  { \
142  GeoType1 geo1(pg1, g1->get_data_size(), g1->get_flags(), \
143  g1->get_srid()); \
144  GeoType2 geo2(pg2, g2->get_data_size(), g2->get_flags(), \
145  g2->get_srid()); \
146  res= boost::geometry::bgfunc(geo1, geo2); \
147  } \
148  else \
149  { \
150  my_error(ER_GIS_INVALID_DATA, MYF(0), "st_" #bgfunc); \
151  (*(pnullval))= 1; \
152  } \
153 } while (0)
