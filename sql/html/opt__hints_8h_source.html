136 class Opt_hints : public Sql_alloc
137 {
138  /*
139  Name of object referred by the hint.
140  This name is empty for global level,
141  query block name for query block level,
142  table name for table level and key name
143  for key level.
144  */
145  const LEX_CSTRING *name;
146  /*
147  Parent object. There is no parent for global level,
148  for query block level parent is Opt_hints_global object,
149  for table level parent is Opt_hints_qb object,
150  for key level parent is Opt_hints_key object.
151  */
152  Opt_hints *parent;
153 
154  Opt_hints_map hints_map; // Hint map
155 
156  /* Array of child objects. i.e. array of the lower level objects */
157  Mem_root_array<Opt_hints*, true> child_array;
158  /* true if hint is connected to the real object */
159  bool resolved;
160  /* Number of resolved children */
161  uint resolved_children;
162 
163 public:
164 
165  Opt_hints(const LEX_CSTRING *name_arg,
166  Opt_hints *parent_arg,
167  MEM_ROOT *mem_root_arg)
168  : name(name_arg), parent(parent_arg), child_array(mem_root_arg),
169  resolved(false), resolved_children(0)
170  { }
171 
172  bool is_specified(opt_hints_enum type_arg) const
173  {
174  return hints_map.is_specified(type_arg);
175  }
176 
187  bool set_switch(bool switch_state_arg,
188  opt_hints_enum type_arg,
189  bool check_parent)
190  {
191  if (is_specified(type_arg) ||
192  (check_parent && parent->is_specified(type_arg)))
193  return true;
194 
195  hints_map.set_switch(type_arg, switch_state_arg);
196  return false;
197  }
198 
207  bool get_switch(opt_hints_enum type_arg) const;
208 
209  virtual const LEX_CSTRING *get_name() const { return name; }
210  void set_name(const LEX_CSTRING *name_arg) { name= name_arg; }
211  Opt_hints *get_parent() const { return parent; }
212  void set_resolved() { resolved= true; }
213  bool is_resolved() const { return resolved; }
214  void incr_resolved_children() { resolved_children++; }
215  Mem_root_array<Opt_hints*, true> *child_array_ptr() { return &child_array; }
216 
217  bool is_all_resolved() const
218  {
219  return child_array.size() == resolved_children;
220  }
221 
222  void register_child(Opt_hints* hint_arg)
223  {
224  child_array.push_back(hint_arg);
225  }

270  virtual void append_name(THD *thd, String *str)= 0;
