112 /*
113  A JSON path expression.
114 
115  From the user's point of view,
116  a path expression is a string literal with the following structure.
117  We parse this structure into a Json_path class:
118 
119  <code><pre>
120 
121  pathExpression ::= scope pathLeg (pathLeg)*
122 
123  scope ::= [ columnReference ] dollarSign
124 
125  columnReference ::=
126  [
127  [ databaseIdentifier period ]
128  tableIdentifier period
129  ]
130  columnIdentifier
131 
132  databaseIdentifier ::= sqlIdentifier
133  tableIdentifier ::= sqlIdentifier
134  columnIdentifier ::= sqlIdentifier
135 
136  pathLeg ::= member | arrayLocation | doubleAsterisk
137 
138  member ::= period (keyName | asterisk)
139 
140  arrayLocation ::=
141  leftBracket
142  (non-negative-integer | asterisk)
143  rightBracket
