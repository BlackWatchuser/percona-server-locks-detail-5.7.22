158  bool do_check(THD *thd, set_var *var)
159  {
160  my_bool fixed= FALSE;
161  longlong v;
162  ulonglong uv;
163  const T *vmin, *vmax;
164 
165  v= var->value->val_int();
166  if (SIGNED) /* target variable has signed type */
167  {
168  if (var->value->unsigned_flag)
169  {
170  /*
171  Input value is such a large positive number that MySQL used an
172  unsigned item to hold it. When cast to a signed longlong, if the
173  result is negative there is "cycling" and this is incorrect (large
174  positive input value should not end up as a large negative value in
175  the session signed variable to be set); instead, we need to pick the
176  allowed number closest to the positive input value, i.e. pick the
177  biggest allowed positive integer.
178  */
179  if (v < 0)
180  uv= max_of_int_range(ARGT);
181  else /* no cycling, longlong can hold true value */
182  uv= (ulonglong) v;
183  }
184  else
185  uv= v;
186  /* This will further restrict with VALID_RANGE, BLOCK_SIZE */
187  var->save_result.ulonglong_value=
188  getopt_ll_limit_value(uv, &option, &fixed);
189  }
190  else
191  {
192  if (var->value->unsigned_flag)
193  {
194  /* Guaranteed positive input value, ulonglong can hold it */
195  uv= (ulonglong) v;
196  }
197  else
198  {
199  /*
200  Maybe negative input value; in this case, cast to ulonglong makes it
201  positive, which is wrong. Pick the closest allowed value i.e. 0.
202  */
203  uv= (ulonglong) (v < 0 ? 0 : v);
204  }
205  var->save_result.ulonglong_value=
206  getopt_ull_limit_value(uv, &option, &fixed);
207  }
208 
209  if (max_var_ptr())
210  {
211  /* check constraint set with --maximum-...=X */
212  if (SIGNED)
213  {
214  longlong max_val= *max_var_ptr();
215  if (((longlong)(var->save_result.ulonglong_value)) > max_val)
216  var->save_result.ulonglong_value= max_val;
217  /*
218  Signed variable probably has some kind of symmetry. Then it's good
219  to limit negative values just as we limit positive values.
220  */
221  max_val= -max_val;
222  if (((longlong)(var->save_result.ulonglong_value)) < max_val)
223  var->save_result.ulonglong_value= max_val;
224  }
225  else
226  {
227  ulonglong max_val= *max_var_ptr();
228  if (var->save_result.ulonglong_value > max_val)
229  var->save_result.ulonglong_value= max_val;
230  }
231  }
232 
233  vmin= static_cast<const T *>
234  (getopt_constraint_get_min_value(option.name, 0, FALSE));
235  vmax= static_cast<const T *>
236  (getopt_constraint_get_max_value(option.name, 0, FALSE));
237 
238  if (SIGNED)
239  {
240  if (vmin && (longlong)var->save_result.ulonglong_value < (longlong)*vmin)
241  var->save_result.ulonglong_value= *vmin;
242  if (vmax && (longlong)var->save_result.ulonglong_value > (longlong)*vmax)
243  var->save_result.ulonglong_value= *vmax;
244  if (vmin && (longlong)var->save_result.ulonglong_value
245  > (longlong)-*vmin)
246  var->save_result.ulonglong_value= -*vmin;
247  if (vmax && (longlong)var->save_result.ulonglong_value
248  < (longlong)-*vmax)
249  var->save_result.ulonglong_value= -*vmax;
250  }
251  else
252  {
253  if (vmin && var->save_result.ulonglong_value < (ulonglong)*vmin)
254  var->save_result.ulonglong_value= *vmin;
255  if (vmax && var->save_result.ulonglong_value > (ulonglong)*vmax)
256  var->save_result.ulonglong_value= *vmax;
257  }
258 
259  return throw_bounds_warning(thd, name.str,
260  var->save_result.ulonglong_value !=
261  (ulonglong)v,
262  var->value->unsigned_flag, v);
263  }
264  bool session_update(THD *thd, set_var *var)
265  {
266  session_var(thd, T)= static_cast<T>(var->save_result.ulonglong_value);
267  return false;
268  }
