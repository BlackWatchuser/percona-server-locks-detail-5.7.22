469  inline void rdlock()
470  {
471  mysql_rwlock_rdlock(&rwlock);
472  assert_no_wrlock();
473 #ifndef DBUG_OFF
474  if (dbug_trace)
475  DBUG_PRINT("info", ("%p.rdlock()", this));
476  my_atomic_add32(&lock_state, 1);
477 #endif
478  }
480  inline void wrlock()
481  {
482  mysql_rwlock_wrlock(&rwlock);
483  assert_no_lock();
484 #ifndef DBUG_OFF
485  if (dbug_trace)
486  DBUG_PRINT("info", ("%p.wrlock()", this));
487  my_atomic_store32(&lock_state, -1);
488 #else
489  is_write_lock= true;
490 #endif
491  }
493  inline void unlock()
494  {
495  assert_some_lock();
496 #ifndef DBUG_OFF
497  if (dbug_trace)
498  DBUG_PRINT("info", ("%p.unlock()", this));
499  int val= my_atomic_load32(&lock_state);
500  if (val > 0)
501  my_atomic_add32(&lock_state, -1);
502  else if (val == -1)
503  my_atomic_store32(&lock_state, 0);
504  else
505  DBUG_ASSERT(0);
506 #else
507  is_write_lock= false;
508 #endif
509  mysql_rwlock_unlock(&rwlock);
510  }
515  inline bool is_wrlock()
516  {
517  assert_some_lock();
518 #ifndef DBUG_OFF
519  return get_state() == -1;
520 #else
521  return is_write_lock;
522 #endif
523  }

535  inline void assert_no_wrlock() const
536  { DBUG_ASSERT(get_state() >= 0); }
538  inline void assert_no_rdlock() const
