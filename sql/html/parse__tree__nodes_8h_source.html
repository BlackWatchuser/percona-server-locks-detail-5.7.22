47 class PT_subselect : public PT_select_lex
48 {
49  typedef PT_select_lex super;
50 
51  POS pos;
52  PT_select_lex *query_expression_body;
53 
54 public:
55  explicit PT_subselect(const POS &pos,
56  PT_select_lex *query_expression_body_arg)
57  : pos(pos), query_expression_body(query_expression_body_arg)
58  {}
59 
60  virtual bool contextualize(Parse_context *pc)
61  {
62  if (super::contextualize(pc))
63  return true;
64 
65  LEX *lex= pc->thd->lex;
66  if (!lex->expr_allows_subselect ||
67  lex->sql_command == (int)SQLCOM_PURGE)
68  {
69  error(pc, pos);
70  return true;
71  }
72  /*
73  we are making a "derived table" for the parenthesis
74  as we need to have a lex level to fit the union
75  after the parenthesis, e.g.
76  (SELECT .. ) UNION ... becomes
77  SELECT * FROM ((SELECT ...) UNION ...)
78  */
79  SELECT_LEX *child= lex->new_query(pc->select);
80  if (child == NULL)
81  return true;
82 
83  Parse_context inner_pc(pc->thd, child);
84  if (query_expression_body->contextualize(&inner_pc))
85  return true;
86 
87  lex->pop_context();
88  pc->select->n_child_sum_items += child->n_sum_items;
89  /*
90  A subselect can add fields to an outer select. Reserve space for
91  them.
92  */
93  pc->select->select_n_where_fields+= child->select_n_where_fields;
94  pc->select->select_n_having_items+= child->select_n_having_items;
95  value= query_expression_body->value;
96  return false;
97  }
98 };

294 template<PT_join_table_type Type>
295 class PT_join_table : public Parse_tree_node
296 {
297  typedef Parse_tree_node super;
298 
299 protected:
300  PT_table_list *tab1_node;
301  POS join_pos;
302  PT_table_list *tab2_node;
303 
304  TABLE_LIST *tr1;
305  TABLE_LIST *tr2;
306 
307 
308 public:
309  PT_join_table(PT_table_list *tab1_node_arg, const POS &join_pos_arg,
310  PT_table_list *tab2_node_arg)
311  : tab1_node(tab1_node_arg), join_pos(join_pos_arg), tab2_node(tab2_node_arg),
312  tr1(NULL), tr2(NULL)
313  {
314  DBUG_ASSERT(dbug_exclusive_flags(JTT_NORMAL | JTT_STRAIGHT | JTT_NATURAL));
315  DBUG_ASSERT(dbug_exclusive_flags(JTT_LEFT | JTT_RIGHT));
316  }
317 
318 #ifndef DBUG_OFF
319  bool dbug_exclusive_flags(unsigned int mask)
320  {
321 #ifdef __GNUC__
322  return __builtin_popcount(Type & mask) <= 1;
323 #else
324  return true;
325 #endif
326  }
327 #endif
328 
329  virtual bool contextualize(Parse_context *pc)
330  {
331  if (super::contextualize(pc) || contextualize_tabs(pc))
332  return true;
333 
334  if (Type & (JTT_LEFT | JTT_RIGHT))
335  {
336  if (Type & JTT_LEFT)
337  tr2->outer_join|= JOIN_TYPE_LEFT;
338  else
339  {
340  TABLE_LIST *inner_table= pc->select->convert_right_join();
341  /* swap tr1 and tr2 */
342  DBUG_ASSERT(inner_table == tr1);
343  tr1= tr2;
344  tr2= inner_table;
345  }
346  }
347 
348  if (Type & JTT_NATURAL)
349  add_join_natural(tr1, tr2, NULL, pc->select);
350 
351  if (Type & JTT_STRAIGHT)
352  tr2->straight= true;
353 
354  return false;
355  }
356 
357 protected:
358  bool contextualize_tabs(Parse_context *pc)
359  {
360  if (tr1 != NULL)
361  return false; // already done
362 
363  if (tab1_node->contextualize(pc) || tab2_node->contextualize(pc))
364  return true;
365 
366  tr1= tab1_node->value;
367  tr2= tab2_node->value;
368 
369  if (tr1 == NULL || tr2 == NULL)
370  {
371  error(pc, join_pos);
372  return true;
373  }
374  return false;
375  }
376 };
