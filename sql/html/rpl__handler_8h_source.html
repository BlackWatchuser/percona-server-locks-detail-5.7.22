48 class Delegate {
49 public:
50  typedef List<Observer_info> Observer_info_list;
51  typedef List_iterator<Observer_info> Observer_info_iterator;
52 
53  int add_observer(void *observer, st_plugin_int *plugin)
54  {
55  int ret= FALSE;
56  if (!inited)
57  return TRUE;
58  write_lock();
59  Observer_info_iterator iter(observer_info_list);
60  Observer_info *info= iter++;
61  while (info && info->observer != observer)
62  info= iter++;
63  if (!info)
64  {
65  info= new Observer_info(observer, plugin);
66  if (!info || observer_info_list.push_back(info, &memroot))
67  ret= TRUE;
68  }
69  else
70  ret= TRUE;
71  unlock();
72  return ret;
73  }
74 
75  int remove_observer(void *observer, st_plugin_int *plugin)
76  {
77  int ret= FALSE;
78  if (!inited)
79  return TRUE;
80  write_lock();
81  Observer_info_iterator iter(observer_info_list);
82  Observer_info *info= iter++;
83  while (info && info->observer != observer)
84  info= iter++;
85  if (info)
86  {
87  iter.remove();
88  delete info;
89  }
90  else
91  ret= TRUE;
92  unlock();
93  return ret;
94  }
95 
96  inline Observer_info_iterator observer_info_iter()
97  {
98  return Observer_info_iterator(observer_info_list);
99  }
100 
101  inline bool is_empty()
102  {
103  DBUG_PRINT("debug", ("is_empty: %d", observer_info_list.is_empty()));
104  return observer_info_list.is_empty();
105  }
106 
107  inline int read_lock()
108  {
109  if (!inited)
110  return TRUE;
111  return mysql_rwlock_rdlock(&lock);
112  }
113 
114  inline int write_lock()
115  {
116  if (!inited)
117  return TRUE;
118  return mysql_rwlock_wrlock(&lock);
119  }
120 
121  inline int unlock()
122  {
123  if (!inited)
124  return TRUE;
125  return mysql_rwlock_unlock(&lock);
126  }
127 
128  inline bool is_inited()
129  {
130  return inited;
131  }
132 
133  Delegate(
134 #ifdef HAVE_PSI_INTERFACE
135  PSI_rwlock_key key
136 #endif
137  )
138  {
139  inited= FALSE;
140 #ifdef HAVE_PSI_INTERFACE
141  if (mysql_rwlock_init(key, &lock))
142  return;
143 #else
144  if (mysql_rwlock_init(0, &lock))
145  return;
146 #endif
147  init_sql_alloc(key_memory_delegate, &memroot, 1024, 0);
148  inited= TRUE;
149  }
150  ~Delegate()
151  {
152  inited= FALSE;
153  mysql_rwlock_destroy(&lock);
154  free_root(&memroot, MYF(0));
155  }
156 
157 private:
158  Observer_info_list observer_info_list;
159  mysql_rwlock_t lock;
160  MEM_ROOT memroot;
161  bool inited;
162 };
