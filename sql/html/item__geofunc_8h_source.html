37 class BG_result_buf_mgr
38 {
39  typedef Prealloced_array<void *, 64> Prealloced_buffers;
40 public:
41  BG_result_buf_mgr() :bg_result_buf(NULL), bg_results(PSI_INSTRUMENT_ME)
42  {
43  }
44 
45  ~BG_result_buf_mgr()
46  {
47  free_intermediate_result_buffers();
48  free_result_buffer();
49  }
50 
51  void add_buffer(void *buf)
52  {
53  bg_results.insert_unique(buf);
54  }
55 
56 
57  void forget_buffer(void *buf)
58  {
59  if (bg_result_buf == buf)
60  bg_result_buf= NULL;
61  bg_results.erase_unique(buf);
62  }
63 
64 
65  /* Free intermediate result buffers accumulated during GIS calculation. */
66  void free_intermediate_result_buffers()
67  {
68  bg_results.erase_unique(bg_result_buf);
69  for (Prealloced_buffers::iterator itr= bg_results.begin();
70  itr != bg_results.end(); ++itr)
71  gis_wkb_raw_free(*itr);
72  bg_results.clear();
73  }
74 
75 
76  // Free the final result buffer, should be called after the result used.
77  void free_result_buffer()
78  {
79  gis_wkb_raw_free(bg_result_buf);
80  bg_result_buf= NULL;
81  }
82 
83 
84  void set_result_buffer(void *buf)
85  {
86  bg_result_buf= buf;
87  bg_results.erase_unique(bg_result_buf);
88  }
89 
90 private:
91  /*
92  Hold data buffer of this set operation's final result geometry which is
93  freed next time val_str is called since it can be used by upper Item nodes.
94  */
95  void *bg_result_buf;
96 
97  /*
98  Result buffers for intermediate set operation results, which are freed
99  before val_str returns.
100  */
101  Prealloced_buffers bg_results;
102 };
