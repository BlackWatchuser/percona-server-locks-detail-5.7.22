187  /*
188  In some cases the parser detects a syntax error and calls
189  LEX::cleanup_lex_after_parse_error() method only after finishing parsing
190  the whole routine. In such a situation sp_head::restore_thd_mem_root() will
191  be called twice - the first time as part of normal parsing process and the
192  second time by cleanup_lex_after_parse_error().
193 
194  To avoid ruining active arena/mem_root state in this case we skip
195  restoration of old arena/mem_root if this method has been already called for
196  this routine.
197  */
198  if (!is_parsing_sp_body())
199  return;
200 
201  thd->free_items();
202  thd->mem_root= m_saved_memroot;
203  thd->free_list= m_saved_free_list;
204 
205  m_saved_memroot= NULL;
206  m_saved_free_list= NULL;

886  else if (m_flags & HAS_SQLCOM_FLUSH)
887  my_error(ER_STMT_NOT_ALLOWED_IN_SF_OR_TRG, MYF(0), "FLUSH");
888 
889  return MY_TEST(m_flags &
890  (CONTAINS_DYNAMIC_SQL|MULTI_RESULTS|HAS_SET_AUTOCOMMIT_STMT|
891  HAS_COMMIT_OR_ROLLBACK|HAS_SQLCOM_RESET|HAS_SQLCOM_FLUSH));
892  }
