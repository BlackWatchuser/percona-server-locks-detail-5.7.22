140 class Find_thd_with_id: public Find_THD_Impl
141 {
142 public:
143  Find_thd_with_id(ulong value, bool daemon_allowed):
144  m_id(value), m_daemon_allowed(daemon_allowed) {}
145  virtual bool operator()(THD *thd)
146  {
147  if (!m_daemon_allowed && thd->get_command() == COM_DAEMON)
148  return false;
149  if (thd->thread_id() == m_id)
150  {
151  mysql_mutex_lock(&thd->LOCK_thd_data);
152  return true;
153  }
154  return false;
155  }
156 private:
157  ulong m_id;
158  bool m_daemon_allowed;
159 };
