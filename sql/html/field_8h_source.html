449 template<bool Is_big_endian>
450 void copy_integer(uchar *to, size_t to_length,
451  const uchar* from, size_t from_length,
452  bool is_unsigned)
453 {
454  if (Is_big_endian)
455  {
456  if (is_unsigned)
457  to[0]= from[0];
458  else
459  to[0]= (char)(from[0] ^ 128); // Reverse the sign bit.
460  memcpy(to + 1, from + 1, to_length - 1);
461  }
462  else
463  {
464  const int sign_byte= from[from_length - 1];
465  if (is_unsigned)
466  to[0]= sign_byte;
467  else
468  to[0]= static_cast<char>(sign_byte ^ 128); // Reverse the sign bit.
469  for (size_t i= 1, j= from_length - 2; i < to_length; ++i, --j)
470  to[i]= from[j];
471  }
472 }
