777 class Join_tab_compare_default :
778  public std::binary_function<const JOIN_TAB*, const JOIN_TAB*, bool>
779 {
780 public:
781  bool operator()(const JOIN_TAB *jt1, const JOIN_TAB *jt2)
782  {
783  // Sorting distinct tables, so a table should not be compared with itself
784  DBUG_ASSERT(jt1 != jt2);
785 
786  if (jt1->dependent & jt2->table_ref->map())
787  return false;
788  if (jt2->dependent & jt1->table_ref->map())
789  return true;
790 
791  const bool jt1_keydep_jt2= jt1->key_dependent & jt2->table_ref->map();
792  const bool jt2_keydep_jt1= jt2->key_dependent & jt1->table_ref->map();
793 
794  if (jt1_keydep_jt2 && !jt2_keydep_jt1)
795  return false;
796  if (jt2_keydep_jt1 && !jt1_keydep_jt2)
797  return true;
798 
799  if (jt1->found_records > jt2->found_records)
800  return false;
801  if (jt1->found_records < jt2->found_records)
802  return true;
803 
804  return jt1 < jt2;
805  }
