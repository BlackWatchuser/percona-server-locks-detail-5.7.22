97  void reverse_record_pointers()
98  {
99  if (m_idx < 2) // There is nothing to swap.
100  return;
101  uchar **keys= get_sort_keys();
102  const longlong count= m_idx - 1;
103  for (longlong ix= 0; ix <= count/2; ++ix)
104  std::swap(keys[ix], keys[count - ix]);
105  }

110  void init_record_pointers()
111  {

114  (void) get_next_record_pointer();
115  reverse_record_pointers();
116  }
