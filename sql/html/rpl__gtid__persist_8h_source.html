197 
198  if (!thd->is_operating_gtid_table_implicitly &&
199  table->lock_type >= TL_WRITE_ALLOW_WRITE &&
200  !strcmp(table->table_name, Gtid_table_access_context::TABLE_NAME.str))
201  {
202  if (thd->get_transaction()->xid_state()->has_state(XID_STATE::XA_ACTIVE))
203  {
204  /*
205  Push an error to client if user is modifying the gtid_executed
206  table explicitly by a XA transaction.
207  */
208  thd->raise_error_printf(ER_ERROR_ON_MODIFYING_GTID_EXECUTED_TABLE,
209  table->table_name);
210  DBUG_RETURN(2);
211  }
212  else
213  {
214  /*
215  Push a warning to client if user is modifying the gtid_executed
216  table explicitly by a non-XA transaction.
217  */
218  thd->raise_warning_printf(ER_WARN_ON_MODIFYING_GTID_EXECUTED_TABLE,
219  table->table_name);
220  DBUG_RETURN(1);
221  }
222  }
223 
224  DBUG_RETURN(0);
225  }
