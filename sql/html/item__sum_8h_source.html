562 class Aggregator_distinct : public Aggregator
563 {
564  friend class Item_sum_sum;
565 
566  /*
567  flag to prevent consecutive runs of endup(). Normally in endup there are
568  expensive calculations (like walking the distinct tree for example)
569  which we must do only once if there are no data changes.
570  We can re-use the data for the second and subsequent val_xxx() calls.
571  endup_done set to TRUE also means that the calculated values for
572  the aggregate functions are correct and don't need recalculation.
573  */
574  bool endup_done;
575 
576  /*
577  Used depending on the type of the aggregate function and the presence of
578  blob columns in it:
579  - For COUNT(DISTINCT) and no blob fields this points to a real temporary
580  table. It's used as a hash table.
581  - For AVG/SUM(DISTINCT) or COUNT(DISTINCT) with blob fields only the
582  in-memory data structure of a temporary table is constructed.
583  It's used by the Field classes to transform data into row format.
584  */
585  TABLE *table;
586 
587  /*
588  An array of field lengths on row allocated and used only for
589  COUNT(DISTINCT) with multiple columns and no blobs. Used in
590  Aggregator_distinct::composite_key_cmp (called from Unique to compare
591  nodes
592  */
593  uint32 *field_lengths;
594 
595  /*
596  Used in conjunction with 'table' to support the access to Field classes
597  for COUNT(DISTINCT). Needed by copy_fields()/copy_funcs().
598  */
599  Temp_table_param *tmp_table_param;
600 
601  /*
602  If there are no blobs in the COUNT(DISTINCT) arguments, we can use a tree,
603  which is faster than heap table. In that case, we still use the table
604  to help get things set up, but we insert nothing in it.
605  For AVG/SUM(DISTINCT) we always use this tree (as it takes a single
606  argument) to get the distinct rows.
607  */
608  Unique *tree;
609 
610  /*
611  The length of the temp table row. Must be a member of the class as it
612  gets passed down to simple_raw_key_cmp () as a compare function argument
613  to Unique. simple_raw_key_cmp () is used as a fast comparison function
614  when the entire row can be binary compared.
615  */
616  uint tree_key_length;
617 
618  enum Const_distinct{
619  NOT_CONST= 0,
627  CONST_NULL,
634  CONST_NOT_NULL
635  } const_distinct;
636 
645  bool use_distinct_values;
646 
647 public:
648  Aggregator_distinct (Item_sum *sum) :
649  Aggregator(sum), table(NULL), tmp_table_param(NULL), tree(NULL),
650  const_distinct(NOT_CONST), use_distinct_values(false) {}
651  virtual ~Aggregator_distinct ();
652  Aggregator_type Aggrtype() { return DISTINCT_AGGREGATOR; }
653 
654  bool setup(THD *);

660  virtual bool arg_is_null(bool use_null_value);

1051 /*
1052  variance(a) =
1053 
1054  = sum (ai - avg(a))^2 / count(a) )
1055  = sum (ai^2 - 2*ai*avg(a) + avg(a)^2) / count(a)
1056  = (sum(ai^2) - sum(2*ai*avg(a)) + sum(avg(a)^2))/count(a) =
1057  = (sum(ai^2) - 2*avg(a)*sum(a) + count(a)*avg(a)^2)/count(a) =
1058  = (sum(ai^2) - 2*sum(a)*sum(a)/count(a) + count(a)*sum(a)^2/count(a)^2 )/count(a) =
1059  = (sum(ai^2) - 2*sum(a)^2/count(a) + sum(a)^2/count(a) )/count(a) =
1060  = (sum(ai^2) - sum(a)^2/count(a))/count(a)
1061 
1062 But, this falls prey to catastrophic cancellation. Instead, use the recurrence formulas
1063 
1064  M_{1} = x_{1}, ~ M_{k} = M_{k-1} + (x_{k} - M_{k-1}) / k newline
1065  S_{1} = 0, ~ S_{k} = S_{k-1} + (x_{k} - M_{k-1}) times (x_{k} - M_{k}) newline
1066  for 2 <= k <= n newline
1067  ital variance = S_{n} / (n-1)
1068 
1069 */
