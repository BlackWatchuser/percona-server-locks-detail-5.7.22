89 /*
90  This enumeration type is used only by the function find_item_in_list
91  to return the info on how an item has been resolved against a list
92  of possibly aliased items.
93  The item can be resolved:
94  - against an alias name of the list's element (RESOLVED_AGAINST_ALIAS)
95  - against non-aliased field name of the list (RESOLVED_WITH_NO_ALIAS)
96  - against an aliased field name of the list (RESOLVED_BEHIND_ALIAS)

587 inline bool is_temporary_table(TABLE_LIST *tl)
588 {
589  if (tl->is_view() || tl->schema_table)
590  return FALSE;
591 
592  if (!tl->table)
593  return FALSE;
594 
595  /*
596  NOTE: 'table->s' might be NULL for specially constructed TABLE
597  instances. See SHOW TRIGGERS for example.
598  */
599 
600  if (!tl->table->s)
601  return FALSE;
602 
603  return tl->table->s->tmp_table != NO_TMP_TABLE;
604 }

675 class No_such_table_error_handler : public Internal_error_handler
676 {
677 public:
678  No_such_table_error_handler()
679  : m_handled_errors(0), m_unhandled_errors(0)
680  {}
681 
682  virtual bool handle_condition(THD *thd,
683  uint sql_errno,
684  const char* sqlstate,
685  Sql_condition::enum_severity_level *level,
686  const char* msg)
687  {
688  if (sql_errno == ER_NO_SUCH_TABLE)
689  {
690  m_handled_errors++;
691  return true;
692  }
693 
694  m_unhandled_errors++;
695  return false;
696  }
697 
702  bool safely_trapped_errors() const
703  {
704  /*
705  If m_unhandled_errors != 0, something else, unanticipated, happened,
706  so the error is not trapped but returned to the caller.
707  Multiple ER_NO_SUCH_TABLE can be raised in case of views.
708  */
709  return ((m_handled_errors > 0) && (m_unhandled_errors == 0));
710  }
711 
712 private:
713  int m_handled_errors;
714  int m_unhandled_errors;
715 };
