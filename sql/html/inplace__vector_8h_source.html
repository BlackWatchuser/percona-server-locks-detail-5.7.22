71  objtype *get_space(size_t index)
72  {
73  DBUG_ASSERT(index <= m_obj_count);
74  size_t arr_id= index / array_size;
75  size_t slot_id= index % array_size;
76  objtype *ptr= NULL;
77 
78  DBUG_ASSERT(arr_id <= m_obj_arrays.size());
79 
80  // Appending a new slot causes appending a new array.
81  if (arr_id == m_obj_arrays.size())
82  {
83  DBUG_ASSERT(slot_id == 0);
84  if (m_outof_mem)
85  return NULL;
86  append_new_array();
87  if (m_outof_mem)
88  return NULL;
89  }
90 
91  ptr= m_obj_arrays[arr_id];
92  ptr+= slot_id;
93  return ptr;
94  }
95 
96  void append_new_array()
97  {
98  if (m_outof_mem)
99  return;
100 
101  void *p= my_malloc(m_psi_key, sizeof(objtype) * array_size, MYF(MY_FAE));
102 
103  try
104  {
105  m_obj_arrays.push_back(static_cast<objtype *>(p));
106  }
107  catch (...)
108  {
109  m_outof_mem= true;
110  my_free(p);
111  }
112  }

187  bool resize(size_t new_size, const objtype &val= objtype())
188  {
189  if (new_size > size())
190  {
191  for (size_t i= size(); i < new_size; i++)
192  {
193  if (push_back(val) == NULL)
194  return true;
195  }
196  }
197  else if (new_size < size())
198  {
199  // Destroy objects at tail.
200  for (size_t i= new_size; i < size(); i++)
201  get_object(i)->~objtype();
202
