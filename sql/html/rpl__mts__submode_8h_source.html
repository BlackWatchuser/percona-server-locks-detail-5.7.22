170  /*
171  LEQ comparison of two logical timestamps follows regular rules for
172  integers. SEQ_UNINIT is regarded as the least value in the clock domain.
173 
174  @param a the lhs logical timestamp value
175  @param b the rhs logical timestamp value
176 
177  @return true when a "<=" b,
178  false otherwise
179  */
180  static bool clock_leq(longlong a, longlong b)
181  {
182  if (a == SEQ_UNINIT)
183  return true;
184  else if (b == SEQ_UNINIT)
185  return false;
186  else
187  return a <= b;
188  }
